Import the default composition from `@vendoai/vendo/server`.

```ts
import type { Principal, ActAs, ComponentCatalog, ComponentRegistry, SecretsProvider, Json, RunId } from "@vendoai/core";
import type { LanguageModel } from "ai";                      // peerDependency (00 conventions)
import type { VendoStore } from "@vendoai/store";
import type { VendoAgent } from "@vendoai/agent";
import type { ActionsRegistry, Connector } from "@vendoai/actions";
import type { VendoGuard, PolicyConfig, Judge } from "@vendoai/guard";
import type { AppsRuntime, SandboxAdapter } from "@vendoai/apps";
import type { AutomationsEngine } from "@vendoai/automations";
import type { HostAuthPreset, HostOAuthAdapter } from "@vendoai/vendo/server";

export function createVendo(config: {
  model: LanguageModel;                       // the one required thing
  auth?: HostAuthPreset;                      // one preset fills principal + actAs + oauth; see /connect/act-as-presets
  principal?: (req: Request) => Promise<Principal | null>;   // escape hatch, mutually exclusive with auth; null → ephemeral anonymous
  store?: VendoStore;                         // default: createStore() (PGlite, .vendo/data)
  sandbox?: SandboxAdapter;                   // e.g. cloudSandbox({ apiKey }), e2bSandbox({ apiKey }); absent → auto-selected from E2B_API_KEY / VENDO_API_KEY, else tree-only apps
  catalog?: ComponentCatalog | ComponentRegistry; // host components exposed to the generation prompt; see /connect/host-components
  connectors?: Connector[];
  actAs?: ActAs;                              // escape hatch, mutually exclusive with auth
  policy?: PolicyConfig;                      // "cautious" | "readonly" | "autopilot" | { file } | { rules }
  judge?: Judge;                              // e.g. vendoAutoJudge({ model }); one import, no shorthand union
  secrets?: SecretsProvider;                  // default envSecrets()
  telemetry?: boolean;                        // wires @vendoai/telemetry
  mcp?: boolean | { baseUrl?: string; remoteAs?: object; federation?: object }; // opens the MCP door; off by default
  oauth?: HostOAuthAdapter;                   // escape hatch, mutually exclusive with auth; required when mcp is true and auth supplies no oauth half
  serverActions?: Record<string, ServerActionHandler>; // generated by `vendo sync`
  agent?: {                                   // chat context controls; see Handler options
    toolOutputCap?: number;                   // default 32000; 0 disables the cap
    maxOutputTokens?: number;
    historyWindow?: number;                   // last N whole messages re-sent per turn
    maxInitialTools?: number;                 // initial tool loadout cap; rest via vendo_tools_search
    maxSteps?: number;                        // agent-loop step cap per turn (default 20)
  };
  sessions?: { ttlMs?: number; sweepIntervalMs?: number }; // anonymous-session lifecycle
  approvals?: { parkedCallTtlMs?: number };   // BYO parked-call expiry; default 60 min, 0 disables
  development?: boolean | { root?: string; out?: string }; // dev-only source capture
}): Vendo;

export interface Vendo {
  handler: (req: Request) => Promise<Response>;   // fetch-style; mount at /api/vendo/[...]
  emit(event: string, payload: Json, principal: Principal): Promise<RunId[]>;   // the host-event seam, re-exposed
  agent: VendoAgent; guard: VendoGuard; apps: AppsRuntime; automations: AutomationsEngine; actions: ActionsRegistry; connections: ConnectionsService; store: VendoStore;
  guardedTools: ToolRegistry;   // the guard-bound registry the BYO tool pack executes through; parks pending-approval calls for wire resume
}
```

When `principal(req)` resolves null, each client gets its own ephemeral session
principal, carried by an opaque httpOnly cookie scoped to the wire base. The
cookie is just a pointer; the session's `vendo_sessions` row is the authority.
Sessions are per client, stored on disk like everything else, survive restarts,
and expire through the idle-TTL sweep.

The composition registers app capability tools in actions and guard-binds every
tool registry passed to agent, apps, and automations.

`handler` is framework-agnostic. `nextVendoHandler(vendo)` adapts it to a
Next.js route module. Mount it at `/api/vendo/[...]` unless the client uses a
different base URL.

`emit(event, payload, principal)` is the host-event seam. It returns the ids of
runs started for enabled automations owned by that principal.

## `eraseStore`

`@vendoai/vendo/server` re-exports `eraseStore(store)`, the sanctioned
deletion path across every store table. It is the only supported way to
delete rows from the append-only `vendo_audit` table.

```ts
import { eraseStore } from "@vendoai/vendo/server";

await eraseStore(vendo.store).bySubject("user_123");
await eraseStore(vendo.store).byApp("app_abc");
```

Each call returns a map of per-table deleted counts. See
[Persistence](/deploy/persistence#erasing-data) for the full semantics.

## BYO-agent subpaths

Two subpath exports build the guarded Vendo tool pack for a host that keeps
its own agent loop. Both are thin shims over one framework-neutral core;
every pack tool executes through `vendo.guardedTools`, the same guard-bound
registry Vendo's own loop uses.

```ts
// @vendoai/vendo/ai-sdk: AI SDK ToolSet, built per request
export function vendoTools(
  vendo: Vendo,
  options: { principal: Principal; sessionId?: string; include?: string[]; exclude?: string[] },
): Promise<ToolSet>;

// @vendoai/vendo/mastra: Mastra createTool shapes for Agent({ tools });
// the principal resolves per call from Mastra's request context
export function vendoMastraTools(
  vendo: Vendo,
  options?: { include?: string[]; exclude?: string[] },
): Promise<Record<string, MastraTool>>;

export const VENDO_PRINCIPAL_KEY = "vendo-principal"; // required per request; missing fails closed
export const VENDO_SESSION_KEY = "vendo-session-id";  // optional audit continuity
```

`include`/`exclude` match final (namespaced) tool names exactly; `exclude`
wins. `@mastra/core` is an optional peer dependency used only by the
`./mastra` subpath. Tool outputs are the versioned envelopes documented in
[Embeds and envelopes](/existing-agents/embeds#the-envelope-contract).
